    document.addEventListener('DOMContentLoaded', () => {
        const data = {
            minyak: {
                name: 'Minyak Goreng',
                modal: 50000000,
                kuantitas: '2.500 Liter',
                hargaBeli: 20000,
                hargaJual: 22000,
                pendapatanKotor: 55000000,
                labaKotor: 5000000,
                unit: 'Liter'
            },
            gula: {
                name: 'Gula Pasir',
                modal: 50000000,
                kuantitas: '5.882 Kg',
                hargaBeli: 8500,
                hargaJual: 12000,
                pendapatanKotor: 70584000,
                labaKotor: 20584000,
                unit: 'Kg'
            },
            operasional: 14000000
        };

        const formatRupiah = (number) => {
            return new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0 }).format(number);
        };
        
        const createTable = (productData) => {
            return `
            <table class="min-w-full text-xs md:text-sm bg-white divide-y divide-gray-200">
                <thead class="bg-gray-50">
                    <tr>
                        <th class="px-4 py-2 text-left font-semibold text-gray-600">Deskripsi</th>
                        <th class="px-4 py-2 text-right font-semibold text-gray-600">Jumlah</th>
                    </tr>
                </thead>
                <tbody class="divide-y divide-gray-200">
                    <tr><td class="px-4 py-2 font-bold text-gray-800" colspan="2">Rencana Anggaran Biaya (RAB)</td></tr>
                    <tr><td class="px-4 py-2 text-gray-600">Modal Pembelian</td><td class="px-4 py-2 text-right font-medium">${formatRupiah(productData.modal)}</td></tr>
                    <tr><td class="px-4 py-2 text-gray-600">Kuantitas Pembelian</td><td class="px-4 py-2 text-right font-medium">${productData.kuantitas}</td></tr>
                    <tr><td class="px-4 py-2 text-gray-600">Harga Beli Satuan</td><td class="px-4 py-2 text-right font-medium">${formatRupiah(productData.hargaBeli)} / ${productData.unit}</td></tr>
                    <tr><td class="px-4 py-2 font-bold text-gray-800" colspan="2">Proyeksi Penjualan</td></tr>
                    <tr><td class="px-4 py-2 text-gray-600">Harga Jual Satuan</td><td class="px-4 py-2 text-right font-medium">${formatRupiah(productData.hargaJual)} / ${productData.unit}</td></tr>
                    <tr><td class="px-4 py-2 text-gray-600">Estimasi Pendapatan</td><td class="px-4 py-2 text-right font-medium">${formatRupiah(productData.pendapatanKotor)}</td></tr>
                    <tr class="bg-green-50"><td class="px-4 py-2 font-bold gojek-green-text">Estimasi Laba Kotor</td><td class="px-4 py-2 text-right font-bold gojek-green-text">${formatRupiah(productData.labaKotor)}</td></tr>
                </tbody>
            </table>
            `;
        };

        const rabMinyakContainer = document.getElementById('rabMinyak');
        const rabGulaContainer = document.getElementById('rabGula');
        rabMinyakContainer.innerHTML = createTable(data.minyak);
        rabGulaContainer.innerHTML = createTable(data.gula);

        const btnMinyak = document.getElementById('btnMinyak');
        const btnGula = document.getElementById('btnGula');

        btnMinyak.addEventListener('click', () => {
            rabMinyakContainer.classList.remove('hidden');
            rabGulaContainer.classList.add('hidden');
            btnMinyak.classList.add('gojek-green', 'text-white');
            btnMinyak.classList.remove('bg-gray-200', 'text-gray-700');
            btnGula.classList.add('bg-gray-200', 'text-gray-700');
            btnGula.classList.remove('gojek-green', 'text-white');
        });
        
        btnGula.addEventListener('click', () => {
            rabGulaContainer.classList.remove('hidden');
            rabMinyakContainer.classList.add('hidden');
            btnGula.classList.add('gojek-green', 'text-white');
            btnGula.classList.remove('bg-gray-200', 'text-gray-700');
            btnMinyak.classList.add('bg-gray-200', 'text-gray-700');
            btnMinyak.classList.remove('gojek-green', 'text-white');
        });

        const totalLabaKotor = data.minyak.labaKotor + data.gula.labaKotor;
        const labaBersih = totalLabaKotor - data.operasional;
        
        const profitContributionCtx = document.getElementById('profitContributionChart').getContext('2d');
        new Chart(profitContributionCtx, {
            type: 'doughnut',
            data: {
                labels: ['Laba Kotor Gula Pasir', 'Laba Kotor Minyak Goreng'],
                datasets: [{
                    data: [data.gula.labaKotor, data.minyak.labaKotor],
                    backgroundColor: ['#00AA13', '#F76B1C'],
                    borderColor: '#FFFFFF',
                    borderWidth: 4
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { position: 'bottom', labels: { font: { size: 12 } } },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                let label = context.label || '';
                                if (label) {
                                    label += ': ';
                                }
                                if (context.parsed !== null) {
                                    label += formatRupiah(context.parsed);
                                }
                                return label;
                            }
                        }
                    }
                },
                cutout: '60%'
            }
        });

        const profitBreakdownCtx = document.getElementById('profitBreakdownChart').getContext('2d');
        new Chart(profitBreakdownCtx, {
            type: 'bar',
            data: {
                labels: ['Alokasi'],
                datasets: [
                    {
                        label: 'Laba Bersih',
                        data: [labaBersih],
                        backgroundColor: '#00AA13',
                        borderColor: '#FFFFFF',
                        borderWidth: 2
                    },
                    {
                        label: 'Biaya Operasional',
                        data: [data.operasional],
                        backgroundColor: '#FDB44B',
                        borderColor: '#FFFFFF',
                        borderWidth: 2
                    }
                ]
            },
            options: {
                indexAxis: 'y',
                responsive: true,
                maintainAspectRatio: false,
                scales: {
                    x: { stacked: true, ticks: { callback: (value) => formatRupiah(value) } },
                    y: { stacked: true, display: false }
                },
                plugins: {
                    legend: { position: 'bottom', labels: { font: { size: 12 } } },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                let label = context.dataset.label || '';
                                if (label) {
                                    label += ': ';
                                }
                                if (context.parsed.x !== null) {
                                    label += formatRupiah(context.parsed.x);
                                }
                                return label;
                            }
                        }
                    }
                }
            }
        });

    });
